weigh = float(input('请填入物品的重量KG：'))
area = input('请选择 01、02、03地区：')
if area == '01':
    if weigh <= 2:
        print('需要支付2元')
    elif weigh > 2:
        weigh_1 = (weigh - 2) * 3 + 2
        print('需要支付{}元'.format(weigh_1))
elif area == '02':
    if weigh <= 2:
        print('需要支付2元')
    elif weigh > 2:
        weigh_1 = (weigh - 2) * 5 + 2
        print('需要支付{}元'.format(weigh_1))
elif area == '03':
    if weigh <= 2:
        print('需要支付2元')
    elif weigh > 2:
        weigh_1 = (weigh - 2) * 4 + 2
        print('需要支付{}元'.format(weigh_1))
